19 def parse_options(argv)
20   # Perform manual parsing and cleanup since option parser raises on unknown options.
21   env_index = argv.index("--environment") || argv.index("-e")
22   if env_index
23     argv.delete_at(env_index)
24     environment = argv.delete_at(env_index).strip
25   end
26   ENV["RAILS_ENV"] = environment || "test"
27 
28   w_index = argv.index("--warnings") || argv.index("-w")
29   $VERBOSE = argv.delete_at(w_index) if w_index
30 end
