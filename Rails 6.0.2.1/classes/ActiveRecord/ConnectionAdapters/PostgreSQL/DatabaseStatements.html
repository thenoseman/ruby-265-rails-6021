130 def exec_insert(sql, name = nil, binds = [], pk = nil, sequence_name = nil)
131   if use_insert_returning? || pk == false
132     super
133   else
134     result = exec_query(sql, name, binds)
135     unless sequence_name
136       table_ref = extract_table_ref_from_insert_sql(sql)
137       if table_ref
138         pk = primary_key(table_ref) if pk.nil?
139         pk = suppress_composite_primary_key(pk)

141       end
142       return result unless sequence_name
143     end
144     last_insert_id_result(sequence_name)
145   end
146 end

 97 def exec_query(sql, name = "SQL", binds = [], prepare: false)
 98   execute_and_clear(sql, name, binds, prepare: prepare) do |result|
 99     types = {}
100     fields = result.fields
101     fields.each_with_index do |fname, i|
102       ftype = result.ftype i
103       fmod  = result.fmod i
104       types[fname] = get_oid_type(ftype, fmod, fname)
105     end
106     ActiveRecord::Result.new(fields, result.values, types)
107   end
108 end
