 80 def listen(conn)
 81   conn.without_reconnect do
 82     original_client = conn.respond_to?(:_client) ? conn._client : conn.client
 83 
 84     conn.subscribe("_action_cable_internal") do |on|
 85       on.subscribe do |chan, count|
 86         @subscription_lock.synchronize do
 87           if count == 1
 88             @raw_client = original_client
 89 
 90             until @when_connected.empty?
 91               @when_connected.shift.call
 92             end
 93           end
 94 
 95           if callbacks = @subscribe_callbacks[chan]
 96             next_callback = callbacks.shift
 97             @event_loop.post(&next_callback) if next_callback
 98             @subscribe_callbacks.delete(chan) if callbacks.empty?
 99           end
100         end
101       end
102 
103       on.message do |chan, message|
104         broadcast(chan, message)
105       end
106 
107       on.unsubscribe do |chan, count|
108         if count == 0
109           @subscription_lock.synchronize do
110             @raw_client = nil
111           end
112         end
113       end
114     end
115   end
116 end
